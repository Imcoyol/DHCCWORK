
开始
    在本地创建仓库后第一件事
    {
        git config --global user.name "Your Name"
        git config --global user.email "email@example.com"
        git config --global core.quotepath false
    }
    第一个文件的添加
    {
        创建文件
        编写文件
        git add fileName
        git commit -m "your some tips"
    }
    查看文件的状态
    {
        git diff
    }
    查看仓库的状态
    {
        git status
    }
    查看提交版本
    {
        git log
        git log --pretty=oneline
        git relog
    }
    当前版本回退到上一个版本
    {
        提示：回退版本是把仓库里的刷新到本地库，不保存当前的东西
        git reset --hard HEAD^
    }
